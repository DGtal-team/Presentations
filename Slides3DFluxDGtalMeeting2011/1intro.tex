\subsection{Orientation}


\begin{frame}[t]

\frametitle{1. Visualisation 3D: objectif}

{\large{Inspiré du mécanisme de \texttt{Board2D} (initialement basé sur \texttt{LibBoard}\footnote{(Copyleft, LGPL) 2007 Sébastien Fourey - GREYC ENSICAEN}) }}

\begin{block}{}
\begin{itemize}
\item Chaque primitive capable de s'auto dessiner. 
\item Vérifier le concept \texttt{CDrawableWithBoard2D} (concept vérifié avec \textit{\textcolor<3>{red}{boost}}).  
\item<4-> Mécanisme de ``flux'' avec l'opérateur \texttt{< \hspace{-0.2cm}<}.
\item<6-> Exportation en différents formats (pdf, eps, fig, gif, png). 
\end{itemize}
\end{block}
