
\section{Structure}



\begin{frame}
  \frametitle{Schéma de structuration}
\small
    \begin{exampleblock}{}
    \centering  Noyau de base
    \end{exampleblock}
    \begin{columns}
      \column{5.5cm}


     \begin{block}{Types et structures de données de base\HH}
      \begin{itemize}
      \item Espace discret, points, vecteurs, algèbre linéaire
      \item Infrastructure logicielle : mécanisme de trace, de
        validation de concepts, ...
      \end{itemize}
     \end{block}


      \column{5.5cm}

     \begin{block}{Représentation des images\HH}
      \begin{itemize}
      \item Mécanisme de \emph{Container} générique
      \item Structures adaptées pour de gros volumes (hashtree,...) 
      \end{itemize}
     \end{block}
      
    \end{columns}



    \begin{exampleblock}{}
    \centering  Modules de base       
    \end{exampleblock}
    \begin{columns}
      \column{5cm}
   \begin{block}{Module topologique\HH}       
      \begin{itemize}
      \item Topologie digitale : ensembles, connexité, bords,
        composantes connexes
      \item Topologie discrète : modèle cellulaire, contours,
        modèle de représentation de régions (Carte Topo/Combi)
      \item Calculs d'invariants n-D
      \end{itemize}
    \end{block}


      \column{5cm}
    \begin{block}{Module géométrique\HH}       
      \begin{itemize}
      
      \item Primitives géométriques (définition, reconnaissance,...) 
      \item Analyse contour : decomposition en primitives,
        estimateurs géométriques, ...
      \item Analyse volumique : transformation en distance, axe
        médian, ...
      \end{itemize}
    \end{block}
      
    \end{columns}
\end{frame}
